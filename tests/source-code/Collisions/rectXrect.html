// define a new iio app
TestRectXRect = function( app, settings ){

  // square width
  var oWidth = 15;
  // square speed
  var oSpeed = .8;

  // reverse the given velocity
  function reverse(o,p){ o[p]*=-1 }

  // reverse x velocity
  function reverseX(o){
    reverse(o.vel,'x')
  }
  // reverse y velocity
  function reverseY(o){ 
    reverse(o.vel,'y');
  }

  // common object properties
  var common = {
    width: oWidth*2,
    bounds: {
      left: {
        bound: 0,
        callback: reverseX
      },
      right: {
        bound: app.width,
        callback: reverseX
      },
      top: {
        bound: 0,
        callback: reverseY
      },
      bottom: {
        bound: app.height,
        callback: reverseY
      },
    }
  }

  // add a red square
  var square0 = app.add(new iio.Rectangle(common,{
    color: 'red',
    vel: [ oSpeed, oSpeed/2 ],
    rVel: 0.02,
    pos: [
      app.width/3,
      app.center.y
    ],
  }));

  // add a blue square
  var square1 = app.add(new iio.Rectangle(common,{
    color: 'iioblue',
    vel: [ -oSpeed, -oSpeed/2 ],
    rVel: -0.02,
    pos: [
      app.width*2/3,
      app.center.y + 4
    ],
  }));

  // add a collision callback
  app.collision( square0, square1,
    function(s0,s1){
      // switch object velocities
      var temp = s0.vel;
      s0.vel = s1.vel;
      s1.vel = temp;
      // reverse rotational velocities
      reverse(s0,'rVel');
      reverse(s1,'rVel');
    });
}

// start the app on an existing canvas
iio.start( TestRectXRect, 'canvasId' );
