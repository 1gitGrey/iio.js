// define a new iio app
TestQuadXQuad = function( app, settings ){

  // quad width
  var oWidth = 15;
  // quad speed
  var oSpeed = .8;

  // reverse the given velocity
  function reverse(o,p){ o[p]*=-1 }

  // reverse x velocity
  function reverseX(o){
    reverse(o.vel,'x')
  }
  // reverse y velocity
  function reverseY(o){ 
    reverse(o.vel,'y');
  }

  // common object properties
  var common = {
    width: oWidth*2,
    bounds: {
      left: {
        bound: 0,
        callback: reverseX,
      },
      right: {
        bound: app.width,
        callback: reverseX,
      },
    }
  }

  // add a red quad
  var square0 = app.add(new iio.Quad(common,{
    color: 'red',
    vel: [ oSpeed, 0 ],
    pos: [
      app.width/3,
      app.center.y,
    ],
  }));

  // add a blue quad
  var square1 = app.add(new iio.Quad(common,{
    color: 'iioblue',
    vel: [ -oSpeed, 0 ],
    pos: [
      app.width*2/3,
      app.center.y + 4,
    ],
  }));

  // add a collision callback
  app.collision( square0, square1,
    function(s0,s1){
      // switch object velocities
      var temp = s0.vel;
      s0.vel = s1.vel;
      s1.vel = temp;
      // reverse rotational velocities
      reverse(s0,'rVel');
      reverse(s1,'rVel');
    });
}

// start the app on an existing canvas
iio.start( TestQuadXQuad, 'canvasId' );
