// define a new iio app
PolyXLine = function( app, settings ){

  // line width
  var oWidth = 8;
  // object length
  var oLength = 50;
  // object speed
  var speed = 1;

  // reverse the given velocity
  function reverse(o,p){ o[p]*=-1 }

  // reverse x velocity
  // and rotational velocity
  function reverseX(o){
    reverse(o.vel,'x');
    reverse(o,'rVel');
  }
  // reverse y velocity
  // and rotational velocity
  function reverseY(o){ 
    reverse(o.vel,'y');
    reverse(o,'rVel');
  }

  // common object properties
  var common = {
    bounds: {
      left: {
        bound: 0,
        callback: reverseX
      },
      right: {
        bound: app.width,
        callback: reverseX
      },
      top: {
        bound: 0,
        callback: reverseY
      },
      bottom: {
        bound: app.height,
        callback: reverseY
      },
    }
  }

  // add a red triangle
  var triangle = app.add(new iio.Polygon(common,{
    color: 'red',
    vel: [ speed, speed/2 ],
    rVel: 0.02,
    pos: [
      app.width/3,
      app.center.y
    ],
    vs: [
      [0, -oLength/2],
      [oLength/2, oLength/2],
      [-oLength/2, oLength/2],
    ],
  }));

  // add a blue line
  var line = app.add(new iio.Line(common,{
    width: oWidth,
    color: 'iioblue',
    vel: [ -speed, -speed/2 ],
    rotation: Math.PI/4,
    rVel: -0.02,
    pos: [
      app.width*2/3,
      app.center.y + 4
    ],
    vs: [
      new iio.Vector(0,-oLength),
      new iio.Vector(0,oLength),
    ]
  }));

  // add a collision callback
  app.collision( triangle, line,
    function(s0,s1){
      // switch object velocities
      var temp = s0.vel;
      s0.vel = s1.vel;
      s1.vel = temp;
      // reverse rotational velocities
      reverse(s0,'rVel');
      reverse(s1,'rVel');
    });
}

// start the app on an existing canvas
iio.start( PolyXLine, 'canvasId' );
