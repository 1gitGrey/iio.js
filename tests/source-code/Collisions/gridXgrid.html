// define a new iio app
TestGridXGrid = function( app, settings ){

  // grid width
  var oWidth = 20;
  // grid speed
  var oSpeed = .8;

  // reverse the given velocity
  function reverse(o,p){ o[p]*=-1 }

  // reverse x velocity
  // and rotational velocity
  function reverseX(o){
    reverse(o.vel,'x');
    reverse(o,'rVel');
  }
  // reverse y velocity
  // and rotational velocity
  function reverseY(o){ 
    reverse(o.vel,'y');
    reverse(o,'rVel');
  }

  // common object properties
  var common = {
    width: oWidth*2,
    // rows
    R: 5,
    // columns
    C: 5,
    bounds: {
      left: {
        bound: 0,
        callback: reverseX
      },
      right: {
        bound: app.width,
        callback: reverseX
      },
      top: {
        bound: 0,
        callback: reverseY
      },
      bottom: {
        bound: app.height,
        callback: reverseY
      },
    }
  }

  // add a red grid
  var grid0 = app.add(new iio.Grid(common,{
    color: 'red',
    vel: [ oSpeed, oSpeed/2 ],
    rVel: 0.02,
    pos: [
      app.width/3,
      app.center.y
    ],
  }));

  // add a blue grid
  var grid1 = app.add(new iio.Grid(common,{
    color: 'iioblue',
    vel: [ -oSpeed, -oSpeed/2 ],
    rVel: -0.02,
    pos: [
      app.width*2/3,
      app.center.y + 4
    ],
  }));

  // add a collision callback
  app.collision( grid0, grid1,
    function(g0,g1){
      // switch object velocities
      var temp = g0.vel;
      g0.vel = g1.vel;
      g1.vel = temp;
      // reverse rotational velocities
      reverse(g0,'rVel');
      reverse(g1,'rVel');
    });
}

// start the app on an existing canvas
iio.start( TestGridXGrid, 'canvasId' );
