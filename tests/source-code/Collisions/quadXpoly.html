// define a new iio app
QuadXPoly = function( app, settings ){

  // object width
  var oWidth = 15;
  // object speed
  var speed = .8;

  // reverse the given velocity
  function reverse(o,p){ o[p]*=-1 }

  // reverse x velocity
  function reverseX(o){
    reverse(o.vel,'x');
  }

  // common object properties
  var common = {
    bounds: {
      left: {
        bound: 0,
        callback: reverseX
      },
      right: {
        bound: app.width,
        callback: reverseX
      },
    }
  }

  // add a red square
  var square = app.add(new iio.Quad(common,{
    width: oWidth*2,
    color: 'red',
    vel: [ speed, 0 ],
    pos: [
      app.width/3,
      app.center.y
    ],
  }));

  // add a blue triangle
  var triangle = app.add(new iio.Polygon(common,{
    color: 'iioblue',
    vel: [ -speed, 0 ],
    rVel: 0.02,
    pos: [
      app.width*2/3,
      app.center.y + 4
    ],
    vs: [
      [0, -oWidth],
      [oWidth, oWidth],
      [-oWidth, oWidth],
    ],
  }));

  // add a collision callback
  app.collision( square, triangle,
    function(s,t){
      // switch object velocities
      var temp = s.vel;
      s.vel = t.vel;
      t.vel = temp;
      // reverse triangles rotational velocity
      reverse(t,'rVel');
    });
}

// start the app on an existing canvas
iio.start( QuadXPoly, 'canvasId' );
