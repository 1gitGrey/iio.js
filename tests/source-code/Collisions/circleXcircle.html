// define a new iio app
TestCircleXCircle = function( app, settings ){
  
  // circle radius
  var oRadius = 20;
  // circle speed
  var oSpeed = .8;

  // reverse the given velocity
  function reverse(o,p){ o[p]*=-1 }

  // reverse x velocity
  function reverseX(o){
    reverse(o.vel,'x')
  }
  // reverse y velocity
  function reverseY(o){ 
    reverse(o.vel,'y');
  }

  // common object properties
  var common = {
    radius: oRadius,
    bounds: {
      left: {
        bound: 0,
        callback: reverseX
      },
      right: {
        bound: app.width,
        callback: reverseX
      },
      top: {
        bound: 0,
        callback: reverseY
      },
      bottom: {
        bound: app.height,
        callback: reverseY
      },
    }
  }

  // add a red circle
  var circle0 = app.add(new iio.Ellipse(common,{
    color: 'red',
    vel: [ oSpeed, oSpeed/2 ],
    pos: [
      app.width/3,
      app.center.y
    ],
  }));

  // add a blue circle
  var circle1 = app.add(new iio.Ellipse(common,{
    color: 'iioblue',
    vel: [ -oSpeed, -oSpeed/2 ],
    pos: [
      app.width*2/3,
      app.center.y + 4
    ],
  }));

  // add a collision callback
  app.collision( circle0, circle1,
    function(c0,c1){
      // switch object velocities
      var temp = c0.vel;
      c0.vel = c1.vel;
      c1.vel = temp;
    });
}

// start the app on an existing canvas
iio.start( TestCircleXCircle, 'canvasId' );
