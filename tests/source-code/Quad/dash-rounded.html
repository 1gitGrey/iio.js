// define a new iio app
TestDashRounded = function( app, settings ){

	// define shared Quad properties
	app.add(new iio.Quad({
		// set position to app center
		pos: app.center.clone(),
		// set dimensions to 30x60px
		width: 30,
		height: 60,
		// set outline color to green
		outline: 'green',
		// set line width to 10px
		lineWidth: 5,
		// set dash with .1px length, 10px spacing
		dash: [ .1, 10 ],
		// a round line cap and small dash length
		// creates a dotted effect
		lineCap: 'round',
		// set a dash offset of 10px
		dashOffset: 10
	}));
}

// start the app on an existing canvas
iio.start( TestDashRounded, 'canvasId' );
