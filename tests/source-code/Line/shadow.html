// define a new iio app
TestShadow = function( app, settings ){

  // set app background color to white
  app.set({ color:'white' })

  // add a new line to the app
  app.add(new iio.Line({
    // set position to app center
    pos: app.center.clone(),
    // set color to green
    color: 'green',
    // set width to 10px
    width: 10,
    // set shadow color 
    // to 50% transparent black
    shadow: new iio.Color( 0,0,0,.5 ),
    // set shadow blur
    shadowBlur: 5,
    // set shadow offset
    shadowOffset: [ 4,4 ],
    // set vertices relative to pos
    vs:[
      // start vertex position
      [ 0,-30 ],
      // end vertex position
      [ 0, 30 ]
    ]
  }));
}

// start the app on an existing canvas
iio.start( TestShadow, 'canvasId' );
