// define a new iio app
TestChild = function( app, settings ){

  // define shared line properties
  var line_props = {
    // set color to blue
    color: 'blue',
    // set width to 10px
    width: 10,
    // set vertices
    vs:[
      // start vertex position
      [ 0,-30 ],
      // end vertex position
      [ 0, 30 ]
    ]
  }

  // add a new line to the app
  // with the shared properties
  app.add(new iio.Line( line_props, {
    // set position to app center
    pos: app.center.clone(),
    // set rotational velocity
    // to .02 radians / update
    rVel: .02

    // add a new line to the line
    // with the shared properties
  })).add(new iio.Line( line_props, {
    // set vertices
    vs:[
      // start vertex position
      [ -30, 0 ],
      // end vertex position
      [  30, 0 ]
    ]
  }));
}

// start the app on an existing canvas
iio.start( TestChild, 'canvasId' );
