// define a new iio app
TestChild = function( app, settings ){

	// define shared circle properties
	var props = {
		// set color to blue
		color: 'blue',
		// set line width to 5px
		lineWidth: 5
	}

	// add a new circle to the app
	// with the shared properties
	app.add(new iio.Circle( props, {
		// set position to app center
		pos: app.center.clone(),
		// set origin relative to pos
		origin: [ 8, -8 ],
		// set radius to 25px
		radius: 25,
		// set rotational velocity
		// to .02 radians / update
		rVel: .02

		// add a new circle to the circle
		// with the shared properties
	})).add(new iio.Circle( props, {
		// set radius to 12px
		radius: 12
	}));
}

// start the app on an existing canvas
iio.start( TestChild, 'canvasId' );
