// define a new iio app
TestGradient = function( app, settings ){

  // add a new triangle to the app
  app.add(new iio.Polygon({
    // set position to app center
    pos: app.center.clone(),
    // set vertices relative to pos
    vs: [
      [   0,-20 ],
      [  20, 20 ],
      [ -20, 20 ]
    ],
    // set color to a new linear gradient
    color: new iio.Gradient({
      // set gradient start position
      start: [ 0, -20 ],
      // set gradient end position
      end: [ 0, 20 ],
      // define color stops from [0,1]
      stops: [
        // at 0% length, show green
        [ 0, 'green' ],
        // at 100% length, show black
        [ 1, 'black' ]
      ]
    })
  }));
}

// start the app on an existing canvas
iio.start( TestGradient, 'canvasId' );
