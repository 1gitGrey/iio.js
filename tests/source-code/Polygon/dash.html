// define a new iio app
TestDash = function( app, settings ){

	// define shared triangle properties
	app.add(new iio.Polygon({
		// set position to app center
		pos: app.center.clone(),
		// set outline color to green
		outline: 'green',
		// set line width to 10px
		lineWidth: 10,
		// set dash with 10px length and spacing
		dash: 10,
		// set vertices relative to pos
		vs: [
			[   0,-20 ],
			[  20, 20 ],
			[ -20, 20 ]
		]
	}));
}

// start the app on an existing canvas
iio.start( TestDash, 'canvasId' );
