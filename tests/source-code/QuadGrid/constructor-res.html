// define a new iio app
TestConstructorRes = function( app, settings ){

  // add a new grid to the app
  app.add(new iio.QuadGrid({
    // set position to app center
    pos: app.center,
    // set color to green
    color: 'green',
    // set line width to 5px
    lineWidth: 5,
    // set number of rows
    R: 3,
    // set number of columns
    C: 3,
    // set cell resolution to 25x25px
    res: 25
  }));
}

// start the app on an existing canvas
iio.start( TestConstructorRes, 'canvasId' );
