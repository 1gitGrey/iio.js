// define a new iio app
TestHidden = function( app, settings ){

	// set app update speed to 1 fps
	app.loop(1);

	// add a new grid to the app
	app.add(new iio.QuadGrid({
		// set position to app center
		pos: app.center,
		// set color to blue
		color: 'blue',
		// set width and height to 80px
		width: 80,
		// set line width to 5px
		lineWidth: 5,
		// set number of rows
		R: 3,
		// set number of columns
		C: 3,
		// initialize hidden property
		hidden: false,
		// set update function
		onUpdate: function(){
			// switch the hidden property
			// true/false
			this.hidden = !this.hidden;
		}
	}));
}

// start the app on an existing canvas
iio.start( TestHidden, 'canvasId' );
