// define a new iio app
TestAlpha = function( app, settings ){

	// add a new grid to the app
	app.add(new iio.QuadGrid({
		// set position to app center
		pos: app.center,
		// set color to green
		color: 'green',
		// set width and height to 80px
		width: 80,
		// set line width to 5px
		lineWidth: 5,
		// set number of rows
		R: 3,
		// set number of columns
		C: 3,
		// define fade animation
		fade: {
			// set fade speed
			// to .03 alpha / update
			speed: .03,
			// add lower bound to prevent
			// full transparency
			lowerBound: .2,
			// callback when alpha = 0 or 1
			callback: function(o){
				// reverse fade direction
				o.fade.speed *= -1;
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestAlpha, 'canvasId' );
