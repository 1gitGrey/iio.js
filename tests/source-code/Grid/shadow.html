// define a new iio app
TestShadow = function( app, settings ){

	// set app background color to white
	app.set({ color:'white' })

	// add a new grid to the app
	app.add(new iio.Grid({
		// set position to app center
		pos: app.center,
		// set color to green
		color: 'green',
		// set width and height to 80px
		width: 80,
		// set line width to 5px
		lineWidth: 5,
		// set number of rows
		R: 3,
		// set number of columns
		C: 3,
		// set dash with 10px width and 4px spacing
		dash: [ 10, 4 ],
		// set line width to 5px
		lineWidth: 5,
		// set shadow color 
		// to 50% transparent black
		shadow: new iio.Color( 0,0,0,.5 ),
		// set shadow blur
		shadowBlur: 5,
		// set shadow offset
		shadowOffset: [ 4,4 ]
	}));
}

// start the app on an existing canvas
iio.start( TestShadow, 'canvasId' );
