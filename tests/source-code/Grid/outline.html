// define a new iio app
TestOutline = function( app, settings ){

	// set app update speed to 15 fps
	app.loop(15);

	// add a new grid to the app
	app.add(new iio.Grid({
		// set position to app center
		pos: app.center,
		// set outline color to blue
		color: 'blue',
		// set width and height to 80px
		width: 80,
		// set line width to 5px
		lineWidth: 5,
		// set number of rows
		R: 3,
		// set number of columns
		C: 3,
		// set line width to 1px
		lineWidth: 1,
		// define new boolean growing property
		growing: true,
		// update the width and color on each update
		onUpdate: function(){
			if(this.growing){
				// increase the line width by 1px
				this.lineWidth++;
				if(this.lineWidth > 20)
					this.growing = false;
			} else {
				// decrease the line width by 1px
				this.lineWidth--;
				if(this.lineWidth < 1)
					this.growing = true;
			}
			switch( this.cycle ){
				case 1: 
					// subtract green if over limit
					if( this.color.g > 100 )
						this.color.g--;
					// subtract red if over limit
					else if( this.color.r > 100 )
						this.color.r--;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
					break;
				case 2: 
					// add blue if under limit
					if( this.color.b < 200 )
						this.color.b++;
					// add red if under limit
					else if( this.color.r < 200 )
						this.color.r++;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
					break;
				case 3: 
					// subtract green if over limit
					if( this.color.g > 0 )
						this.color.g--;
					// subtract red if over limit
					else if( this.color.r > 0 )
						this.color.r--;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
					break;
				default: 
					// add red if under limit
					if( this.color.r < 255 )
						this.color.r++;
					// add blue if under limit
					else if( this.color.b < 255 )
						this.color.b++;
					// randomize cycle
					else this.cycle = iio.randomInt( 1,3 );
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestOutline, 'canvasId' );
