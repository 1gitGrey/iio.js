TestWidth = function( app, settings ){

	app.loop(15);

	app.add(new iio.Line({
		pos: app.center.clone(),
		color: 'green',
		width: 1,
		vs:[
			[ -30,-30 ],
			[  30, 30 ]
		],
		growing: true,
		onUpdate: function(){
			if( this.growing ){
			this.width++;
			if(this.width > 30)
				this.growing = false;
		} else {
			this.width--;
			if( this.width < 2 )
				this.growing = true;
		}
	}
	}));
}

iio.start( TestWidth, 'canvasId' );
