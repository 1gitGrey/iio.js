TestAccs = function( app, settings ){

	var speed = 0.5;

	var checkBound = function(o,v){
		if( o.vs[v].x > app.width/2 )
			o.vels[v].x = -1;
		else if ( o.vs[v].x < -app.width/2 )
			o.vels[v].x = 1;
	}

	app.add(new iio.Line({
		pos: app.center.clone(),
		color: 'green',
		width: 10,
		vs:[
			[ 0, 30 ],
			[ 0,-30 ]
		],
		vels:[
			[  speed, 0 ],
			[ -speed, 0 ]
		],
		accs:[
			[  .01, 0 ],
			[ -.01, 0 ]
		],
		onUpdate: function(){
			checkBound( this,0 );
			checkBound( this,1 );
		}
	}));
}

iio.start( TestAccs, 'canvasId' );
