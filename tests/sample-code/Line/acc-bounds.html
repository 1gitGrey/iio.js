TestAccBounds = function( app, settings ){

	var speed = 0.4;

	app.add(new iio.Line({
		pos: app.center.clone(),
		color: 'green',
		width: 10,
		vel: [ speed, 0 ],
		acc: [ .01, 0 ],
		vs:[
			[ 0,-30 ],
			[ 0, 30 ]
		],
		bounds: {
			right: {
				bound: app.center.x+15, 
				callback: function(o){
					o.vel.x = -.5
				} 
			},
			left: {
				bound: app.center.x-15, 
				callback: function(o){
					o.vel.x = .5
				} 
			}
		}
	}));
}

iio.start( TestAccBounds, 'canvasId' );
