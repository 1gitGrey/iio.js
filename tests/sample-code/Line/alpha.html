// define a new iio app
TestAlpha = function( app, settings ){

	// add a new line to the app
	app.add(new iio.Line({
		// set position to app center
		pos: app.center.clone(),
		// set vertices relative to pos
		vs:[
			// start vertex position
			[ -30,-30 ],
			// end vertex position
			[  30, 30 ]
		],
		// set color to green
		color: 'green',
		// set width to 10px
		width: 10,
		// define fade animation
		fade: {
			// set fade speed
			// to .03 alpha / update
			speed: .03,
			// callback when alpha = 0 or 1
			callback: function(o){
				// reverse fade direction
				o.fade.speed *= -1;
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestAlpha, 'canvasId' );
