TestVelBounds = function( app, settings ){

	var speed = 0.4;

	function reverse(o){ o.vel.x *= -1 }

	app.add(new iio.Line({
		pos: app.center.clone(),
		color: 'blue',
		width: 10,
		vel: [ speed, 0 ],
		vs:[
			[ 0,-30 ],
			[ 0, 30 ]
		],
		bounds: {
			right: {
				bound: app.center.x+15, 
				callback: reverse
			},
			left: {
				bound: app.center.x-15, 
				callback: reverse
			}
		}
	}));
}

iio.start( TestVelBounds, 'canvasId' );
