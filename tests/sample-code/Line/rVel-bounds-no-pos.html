// define a new iio app
TestrVelBoundsNoPos = function( app, settings ){

	// reverse rotational velocity
	function reverse(o){ o.rVel *= -1 }

	// add a new line to the app
	app.add(new iio.Line({
		// set color to blue
		color: 'blue',
		// set width to 10px
		width: 10,
		// set rotational velocity
		// to .02 radians / update
		rVel: .02,
		// set rotation origin
		origin: app.center.clone(),
		// set vertices relative to app 0,0
		vs:[
			// start vertex position
			[ 20, 20 ],
			// end vertex position
			[ app.width-20, app.height-20 ]
		],
		// set bounds for movement
		bounds: {
			// define rotation upper limit and callback
			rightRotation: {
				bound: Math.PI/2,
				callback: reverse
			},
			// define rotation lower limit and callback
			leftRotation: {
				bound: 0,
				callback: reverse
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestrVelBoundsNoPos, 'canvasId' );
