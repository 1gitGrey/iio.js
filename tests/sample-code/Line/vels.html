// define a new iio app
TestVels = function( app, settings ){

	// define a speed variable
	var speed = 0.5;

	// reverse the velocity of the given vertex
	// if its position is outside the app screen
	// vs are relative to pos
	var checkBound = function( o,v ){
		if( o.vs[v].x > app.width/2 
		 || o.vs[v].x < -app.width/2 )
		 	// reverse x velocity
			o.vels[v].x *= -1;
	}

	// add a new line to the app
	app.add(new iio.Line({
		// set position to app center
		pos: app.center.clone(),
		// set color to blue
		color: 'blue',
		// set width to 10px
		width: 10,
		// set vertices relative to pos
		vs:[
			// start vertex position
			[ 0,-30 ],
			// end vertex position
			[ 0, 30 ]
		],
		// set vertex x velocities
		vels:[
			// start vertex velocity
			[  speed, 0 ],
			// end vertex velocity
			[ -speed, 0 ]
		],
		// check the position of each
		// vertex upon every update
		onUpdate: function(){
			// check start vertex position
			checkBound( this,0 );
			// check end vertex position
			checkBound( this,1 );
		}
	}));
}

// start the app on an existing canvas
iio.start( TestVels, 'canvasId' );
