TestHidden = function( app, settings ){

	app.loop(1);

	app.add(new iio.Line({
		pos: app.center.clone(),
		color: 'blue',
		width: 10,
		hidden: false,
		vs:[
			[ -30,-30 ],
			[  30, 30 ]
		],
		onUpdate: function(){
			this.hidden = !this.hidden;
		}
	}));
}

iio.start( TestHidden, 'canvasId' );
