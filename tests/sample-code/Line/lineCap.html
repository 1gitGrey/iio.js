// define a new iio app
TestLineCap = function( app, settings ){

	// define shared line properties
	var line_props = {
		// set width to 8px
		width: 8,
		// set color to green
		color: 'green',
		// set vertices
		vs:[
			// start vertex position
			[ 0,-30 ],
			// end vertex position
			[ 0, 30 ]
		]
	}

	// add a new line to the app
	// with the shared properties
	app.add(new iio.Line( line_props, {
		// set position 
		pos: [ 
			// pos.x
			app.center.x - line_props.width*2, 
			// pos.y
			app.center.y 
		],
		// set line cap
		lineCap: 'butt',
	}));

	// add a new line to the app
	// with the shared properties
	app.add(new iio.Line( line_props, {
		// set position to app center
		pos: app.center.clone(),
		// set line cap
		lineCap: 'round'
	}));

	// add a new line to the app
	// with the shared properties
	app.add(new iio.Line( line_props, {
		// set position 
		pos: [ 
			// pos.x
			app.center.x + line_props.width*2,
			// pos.y
			app.center.y 
		],
		// set line cap
		lineCap: 'square'
	}));
}

// start the app on an existing canvas
iio.start( TestLineCap, 'canvasId' );
