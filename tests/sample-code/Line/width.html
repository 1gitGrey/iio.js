// define a new iio app
TestWidth = function( app, settings ){

	// set appupdate speed to 15 fps
	app.loop(15);

	// add a new line to the app
	app.add(new iio.Line({
		// set position to app center
		pos: app.center.clone(),
		// set color to green
		color: 'green',
		// set width to 1px
		width: 1,
		// set vertices relative to pos
		vs:[
			// start vertex position
			[ -30,-30 ],
			// end vertex position
			[  30, 30 ]
		],
		// define new boolean property
		growing: true,
		// define update function
		onUpdate: function(){

			if( this.growing ){
				// add 1px to width
				this.width++;
				// reverse growing if width > 30px
				if(this.width > 30)
					this.growing = false;
			} else {
				// subtract 1px from width
				this.width--;
				// reverse growing if width < 30px
				if( this.width < 2 )
					this.growing = true;
			}
		}
	}));
}

// start the app on an existing canvas
iio.start( TestWidth, 'canvasId' );
