// define a new iio app
TestChild = function( app, settings ){

  // define shared properties
  var props = {
    // set number of rows to 3
    R: 3,
    // set number of columns to 3
    C: 3,
    // set total width to 80px
    width: 80,
    // set line width to 5px
    lineWidth: 5
  }

  // add a new grid to the app
  // width the shared properties
  app.add(new iio.Grid( props, {
    // set position to app center
    pos: app.center,
    // set the color to blue
    color: 'blue',
    // set rotational velocity
    // to .02 radians / update
    rVel: .02

    // add a new grid to the grid
    // with the shared properties
  })).add(new iio.Grid( props, {
    // set the rotation to pi/4 radians
    rotation: Math.PI/4,
    // set the color
    color: 'red'
  }));
}

// start the app on an existing canvas
iio.start( TestChild, 'canvasId' );
