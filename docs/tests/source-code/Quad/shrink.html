// define a new iio app
TestAlpha = function( app, settings ){

  // add a new quad to the app
  app.add(new iio.Quad({
    // set position to app center
    pos: app.center.clone(),
    // set color to green
    color: 'green',
    // set radius to 25px
    width: 30,
    height: 60,
    // set shrink animation
    shrink: {
      // shrink speed px/update
      speed: .03,
      // highest radius value
      upperBound: 30,
      // lowest radius value
      lowerBound: 4,
      // callback when radius is past limit
      callback: function(o){
        if( o.radius < o.shrink.lowerBound )
          // reverse the shrink speed (grow)
          o.shrink.speed = -.03;
          // reset the shrink speed (shrink)
        else o.shrink.speed = .03;
      }
    }
  }));
}

// start the app on an existing canvas
iio.start( TestAlpha, 'canvasId' );
