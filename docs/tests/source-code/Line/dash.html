// define a new iio app
TestDash = function( app, settings ){

  // define shared line properties
  var line_props = {
    // set width to 10px
    width: 10,
    // set color to blue
    color: 'blue',
    // set vertices
    vs:[
      // start vertex position
      [ 0,-30 ],
      // end vertex position
      [ 0, 30 ]
    ]
  }

  // add a new line to the app
  // with the shared properties
  app.add(new iio.Line( line_props, {
    // set position
    pos: [ 
      // pos.x
      app.center.x - line_props.width*2,
      // pos.y
      app.center.y 
    ],
    // set dash length and spacing to 10px
    dash: 10
  }));

  // add a new line to the app
  // with the shared properties
  app.add(new iio.Line( line_props, {
    // set position to app center
    pos: app.center,
    // set dash length to .1px and spacing to 15px
    dash: [ .1, 15 ],
    // set dash offset to 10px
    dashOffset: 10,
    // set line cap
    lineCap: 'round'
  }));

  // add a new line to the app
  // with the shared properties
  app.add(new iio.Line( line_props, {
    // set position
    pos: [ 
      // pos.x
      app.center.x + line_props.width*2,
      // pos.y
      app.center.y 
    ],
    // set dash length to 1px and spacing to 3px
    dash: [ 1, 3 ]
  }));
}

// start the app on an existing canvas
iio.start( TestDash, 'canvasId' );
