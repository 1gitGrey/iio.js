// define a new iio app
TestOrigin = function( app, settings ){

  // add a new line to the app
  app.add(new iio.Line({
    // set position
    pos: [ 30, 30 ],
    // set origin of rotation
    origin: [ 15, 15 ],
    // set color to green
    color: 'green',
    // set width to 10px
    width: 10,
    // set rotational velocity
    // to .02 radians / update
    rVel: .02,
    // set vertices relative to pos
    vs:[
      // start vertex position
      [ -15,-15 ],
      // end vertex position
      [  15, 15 ]
    ]
  }));
}

// start the app on an existing canvas
iio.start( TestOrigin, 'canvasId' );
