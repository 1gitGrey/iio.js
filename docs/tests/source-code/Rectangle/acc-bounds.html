// define a new iio app
TestAccBounds = function( app, settings ){

  // define a speed variable
  var speed = 0.5;

  // add a new rectangle to the app
  app.add(new iio.Rectangle({
    // set position to app center
    pos: app.center.clone(),
    // set color to green
    color: 'green',
    // set dimensions to 30x60px
    width: 30,
    height: 60,
    // set x velocity
    vel: [ speed, 0 ],
    // set x acceleration
    acc: [ .01, 0 ],
    // set bounds for movement
    bounds: {
      // define pos.x upper limit and callback
      right: {
        bound: app.width, 
        callback: function(o){
          // set the x velocity
          o.vel.x = -speed*2;
        } 
      }
    }
  }));
}

// start the app on an existing canvas
iio.start( TestAccBounds, 'canvasId' );
